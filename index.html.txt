// compare function
function initCompare(overlay,line){
let down=false;
line.addEventListener('mousedown',()=>down=true);
document.addEventListener('mouseup',()=>down=false);
document.addEventListener('mousemove',e=>{
if(!down) return;
let rect=overlay.parentElement.getBoundingClientRect();
let x=e.clientX-rect.left;
let p=(x/rect.width)*100;
overlay.style.width=p+'%';
line.style.left=p+'%';
});
}
